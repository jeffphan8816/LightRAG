rm -rf /tmp/lambda-zip-build /tmp/lambda-deploy.zip
mkdir -p /tmp/lambda-zip-build

uv pip install --no-cache \
  --target /tmp/lambda-zip-build \
  --python-platform aarch64-unknown-linux-gnu \
  --python-version 3.14 \
  --only-binary=:all: \
  --no-binary=lightrag-retriever \
  "./lightrag_retriever[all]"

TIKTOKEN_CACHE_DIR=/tmp/lambda-zip-build/tiktoken \
  python -c "import tiktoken; tiktoken.get_encoding('cl100k_base')"

cp deploy/lambda/handler.py /tmp/lambda-zip-build/

cd /tmp/lambda-zip-build
zip -r -q /tmp/lambda-deploy.zip .

aws lambda update-function-configuration \
  --function-name lightrag-retriever \
  --runtime python3.14

aws lambda wait function-updated --function-name lightrag-retriever

aws lambda update-function-code \
  --function-name lightrag-retriever \
  --zip-file fileb:///tmp/lambda-deploy.zip

aws lambda wait function-updated --function-name lightrag-retriever
echo '{"query":"List all listing types of Kevadia","mode":"local","top_k":20}' > /tmp/lambda-payload.json
aws lambda invoke \
  --function-name lightrag-retriever \
  --cli-read-timeout 90 \
  --payload fileb:///tmp/lambda-payload.json \
  /tmp/lambda-response.json
cat /tmp/lambda-response.json | python3 -m json.tool
